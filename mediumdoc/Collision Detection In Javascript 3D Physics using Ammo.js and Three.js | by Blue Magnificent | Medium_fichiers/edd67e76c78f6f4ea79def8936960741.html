function handleKeyDown(event){
let keyCode = event.keyCode;
switch(keyCode){
case 87: //W: FORWARD
moveDirection.forward = 1;
break;
case 83: //S: BACK
moveDirection.back = 1;
break;
case 65: //A: LEFT
moveDirection.left = 1;
break;
case 68: //D: RIGHT
moveDirection.right = 1;
break;
case 84://T
checkContact();
break;
case 74://J
jump();
break;
}
}
